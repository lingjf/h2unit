CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

	
PROJECT(h2_unit)

INCLUDE_DIRECTORIES(..)

SET(CMAKE_CXX_COMPILER "g++")

SET(CMAKE_CXX_FLAGS " ${CMAKE_CXX_FLAGS} -std=c++11 -fpermissive -g -O0 -Wall ")

ADD_DEFINITIONS(-DH2UNIT_VERSION="dev")

AUX_SOURCE_DIRECTORY (. SRC_TEST)

ADD_CUSTOM_TARGET(syntax ALL
    COMMAND cp ${CMAKE_CURRENT_SOURCE_DIR}/test_syntax1.cpp ${CMAKE_CURRENT_SOURCE_DIR}/test_syntax2.cpp
    DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/test_syntax1.cpp
)

ADD_EXECUTABLE(a.out ${SRC_TEST})

ADD_DEPENDENCIES(a.out syntax)
