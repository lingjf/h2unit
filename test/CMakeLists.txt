CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT(h2_unit)

SET(CMAKE_CXX_FLAGS " ${CMAKE_CXX_FLAGS} -std=c++11 -g")

IF(CMAKE_SYSTEM_NAME STREQUAL "Darwin")
    SET(CMAKE_EXE_LINKER_FLAGS " ${CMAKE_EXE_LINKER_FLAGS} -Wl,-segprot,__TEXT,rwx,rwx")
ENDIF()

ADD_CUSTOM_TARGET(release ALL
    COMMAND  cd ${CMAKE_CURRENT_SOURCE_DIR}/../build && python concatenate.py
)

AUX_SOURCE_DIRECTORY(. SRC_TEST)

ADD_EXECUTABLE(a.out ${SRC_TEST})
